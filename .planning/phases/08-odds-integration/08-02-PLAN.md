```typescript
const REQUEST_TIMEOUT_MS = 15_000;  // use 30_000 for Odds API (slower than NBA CDN)

async function nbaGet&lt;T&gt;(path: string): Promise&lt;T&gt; {
  const controller = new AbortController();
  const timeout = setTimeout(() => controller.abort(), REQUEST_TIMEOUT_MS);
  try {
    const res = await fetch(`${NBA_CDN}${path}`, { signal: controller.signal });
    if (!res.ok) throw new Error(`NBA CDN ${res.status}: ${path}`);
    return res.json() as Promise&lt;T&gt;;
  } catch (err) {
    if ((err as Error).name === 'AbortError') throw new Error(`TIMEOUT...`);
    throw err;
  } finally {
    clearTimeout(timeout);
  }
}
```
