From scripts/lib/poll-live-logic.ts — calculateBackoff (already tested, confirm passing):
```typescript
export function calculateBackoff(failureCount: number, baseMs: number): number {
  return Math.min(baseMs * Math.pow(2, failureCount), 60_000);
}
// failureCount=0, base=12000 → 12000
// failureCount=3, base=12000 → min(12000*8, 60000) → 60000
// failureCount=10, base=12000 → capped at 60000
```
