    Replace the stub at `app/players/page.tsx` with an async Server Component.

    ```typescript
    // app/players/page.tsx
    import { RosterViewToggle } from '@/src/components/players/RosterViewToggle'
    // NOTE: check if components/ maps to @/components/ or @/src/components/ per project tsconfig
    // Phase 12 decision: @/* maps to project root, src/ is a subdirectory
    // components/ is at root level (not inside src/) — use @/components/players/RosterViewToggle

    export default async function PlayersPage() {
      const baseUrl = process.env.NEXT_PUBLIC_BASE_URL ?? 'http://localhost:3000'
      let players: Player[] = []
      try {
        const res = await fetch(`${baseUrl}/api/players?include_traded=true`, { cache: 'no-store' })
        if (res.ok) {
          const data = await res.json()
          players = data.players ?? []
        }
      } catch {
        // Graceful degradation — render empty roster, don't throw
      }

      return (
        <div className="px-6 py-6 max-w-[1440px] mx-auto space-y-4">
          <div>
            <h1 className="text-2xl font-semibold text-foreground">Players</h1>
            <p className="text-sm text-muted-foreground mt-1">
              Clippers roster — {new Date().getFullYear()}–{(new Date().getFullYear() + 1).toString().slice(2)} season
            </p>
          </div>
          <RosterViewToggle players={players} />
        </div>
      )
    }
    ```
